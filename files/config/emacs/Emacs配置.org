(require 'auth-source)

(defun password-ref (host user)
  (let ((match (car (auth-source-search
                     :host host
                     :user user
                     :require '(:secret)))))
    (if match
	(let ((password (if (functionp (plist-get match :secret))
                            (funcall (plist-get match :secret))
                          (plist-get match :secret))))
          password)
      (user-error "错误，未找到密码。"))))

** 图片处理
#+begin_src emacs-lisp :noweb-ref config
(defun lower-pic (file x y output)
  "调用 FFmpeg 根据给定的x和y轴比例缩小图片文件。"
  (interactive "f选择图片: \nn输入x缩放比例: \nn输入y缩放比例:  \nF指定输出文件名(默认<input>_lower.png): ")
  (let* ((out (if (file-directory-p output) (concat (file-name-sans-extension file) "_lower.png") output))
	 (cmd (format "ffmpeg -i %s -vf \"scale=iw/%d:ih/%d:flags=neighbor\" %s" (shell-quote-argument file) x y (shell-quote-argument out))))
    (call-process-shell-command cmd nil 0)
    (message (format "已保存为%s" (file-name-nondirectory out)))))
#+end_src

  (defun delete-current-file ()
    "Delete the file visited by the current buffer and close the buffer."
    (interactive)
    (let ((file (buffer-file-name)))
      (if (and file (file-exists-p file))
          (when (yes-or-no-p (format "Are you sure you want to delete %s? " file))
            (delete-file file)
            (kill-buffer (current-buffer))
            (message "Deleted file: %s" file))
        (message "No file is associated with this buffer."))))

  (global-set-key (kbd "C-c D") 'delete-current-file)

    "在另一个窗口智能打开 *cargo-eshell*，并运行 cargo run。"
    (interactive)
    (let* ((buf-name "*cargo-eshell*")
  	 (default-directory (file-name-directory (or buffer-file-name default-directory)))
  	 (buf (or (get-buffer buf-name)
                    (save-window-excursion
                      (let ((b (eshell "new")))
                        (with-current-buffer b
  			(rename-buffer buf-name))
                        b)))))
      (display-buffer buf
                      '((display-buffer-reuse-window
  		       display-buffer-use-some-window
                         display-buffer-pop-up-window)
  		      (inhibit-same-window . t)))
      (with-current-buffer buf
        (goto-char (point-max))
        (eshell-send-eof-to-process)
        (insert "cargo run")
        (eshell-send-input))))

(defun chiko/openwith-inhibit-hack (old-fun operation &rest args)
  "只有特定条件满足才会执行正常逻辑"
  (let* ((file (car args))
	 (file-size (if (and (stringp file) (file-exists-p file))
                        (file-attribute-size (file-attributes file))
                      0)))
    (if (or (< file-size (* 5 1024 1024))
	    (string-prefix-p "*preview-temp*" (buffer-name))
	    (string-match-p "dirvish" (or file "")))
	(let ((inhibit-file-name-handlers
               (cons 'openwith-file-handler
  		     (and (eq inhibit-file-name-operation operation)
  			  inhibit-file-name-handlers)))
              (inhibit-file-name-operation operation))
  	  (apply operation args))
      (apply old-fun operation args))))

(advice-add 'openwith-file-handler :around #'chiko/openwith-inhibit-hack)
